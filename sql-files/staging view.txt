----------creating view for staging purpose------------

CREATE OR REPLACE VIEW behavior_db.staging_sales AS
SELECT
    TRIM(invoice_no)        AS invoice_id,
    TRIM(stock_code)        AS product_code,
    TRIM(description)       AS product_description,
    quantity,
    unit_price,
    TRIM(customer_id)       AS customer_id,
    country,
    TRY_CAST(NULLIF(TRIM(invoice_date), '') AS TIMESTAMP) AS invoice_timestamp,
    CAST(
        TRY_CAST(NULLIF(TRIM(invoice_date), '') AS TIMESTAMP)
    AS DATE) AS invoice_date
FROM raw_sales;
